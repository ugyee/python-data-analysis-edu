🛒 项目五：购物篮分析
进阶

使用Apriori算法分析用户购买商品的关联规则，发现“购买A商品的用户也会购买B”的规律。

关联规则 电商推荐
涉及函数
TransactionEncoder apriori() association_rules() 
import pandas as pd
from mlxtend.frequent_patterns import apriori, association_rules
from mlxtend.preprocessing import TransactionEncoder

df = pd.read_csv('basket.csv')
print(f"原始数据: {len(df)} 条订单")

transactions = df.groupby('order_id')['product_name'].apply(list).tolist()

te = TransactionEncoder()
te_ary = te.fit(transactions).transform(transactions)
basket_df = pd.DataFrame(te_ary, columns=te.columns_)

frequent_itemsets = apriori(basket_df, min_support=0.01, use_colnames=True)
print(f"\n频繁项集数量: {len(frequent_itemsets)}")

rules = association_rules(frequent_itemsets, metric="lift", min_threshold=1.0)
rules = rules.sort_values('lift', ascending=False)

print("\nTop 10 关联规则：")
print(rules[['antecedents', 'consequents', 'support', 'confidence', 'lift']].head(10))

rules.to_csv('association_rules.csv', index=False)
print("\n关联规则已保存到 association_rules.csv")
👥 项目六：客户聚类分析
进阶

使用K-Means算法对客户进行分群，分析不同客户群体的消费特征和行为模式。

用户分群 聚类算法
涉及函数
KMeans() StandardScaler() silhouette_score() 
import pandas as pd
import numpy as np
from sklearn.cluster import KMeans
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import silhouette_score
import matplotlib.pyplot as plt

df = pd.read_csv('customer_features.csv')
features = ['total_spend', 'order_count', 'avg_order_value', 'days_since_last_order']
X = df[features].fillna(0)

scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

inertias = []
silhouette_scores = []
K_range = range(2, 10)

for k in K_range:
    kmeans = KMeans(n_clusters=k, random_state=42, n_init=10)
    kmeans.fit(X_scaled)
    inertias.append(kmeans.inertia_)
    silhouette_scores.append(silhouette_score(X_scaled, kmeans.labels_))

optimal_k = K_range[np.argmax(silhouette_scores)]
print(f"最优聚类数: {optimal_k}")

kmeans = KMeans(n_clusters=optimal_k, random_state=42, n_init=10)
df['cluster'] = kmeans.fit_predict(X_scaled)

cluster_summary = df.groupby('cluster')[features].mean()
print("\n各聚类特征均值：")
print(cluster_summary)

df.to_csv('customer_clustered.csv', index=False)
print(f"\n聚类结果已保存，共 {len(df)} 个客户分为 {optimal_k} 个群体")
🧪 项目七：A/B测试分析
进阶

使用统计假设检验方法分析A/B测试结果，判断新功能是否带来显著提升。

统计检验 转化率分析
涉及函数
ttest_ind() chi2_contingency() confidence_interval() 
import pandas as pd
import numpy as np
from scipy import stats

df = pd.read_csv('ab_test.csv')

control = df[df['group'] == 'control']
treatment = df[df['group'] == 'treatment']

print("样本量：")
print(f"对照组: {len(control)}, 实验组: {len(treatment)}")

control_rate = control['converted'].mean()
treatment_rate = treatment['converted'].mean()
print(f"\n转化率：")
print(f"对照组: {control_rate:.4f} ({control_rate*100:.2f}%)")
print(f"实验组: {treatment_rate:.4f} ({treatment_rate*100:.2f}%)")

lift = (treatment_rate - control_rate) / control_rate
print(f"\n相对提升: {lift:.4f} ({lift*100:.2f}%)")

contingency_table = pd.crosstab(df['group'], df['converted'])
chi2, p_value, dof, expected = stats.chi2_contingency(contingency_table)
print(f"\n卡方检验：")
print(f"χ² = {chi2:.4f}, p-value = {p_value:.4f}")

alpha = 0.05
if p_value < alpha:
    print(f"结论: p < {alpha}，差异显著，拒绝原假设")
else:
    print(f"结论: p >= {alpha}，差异不显著")

se = np.sqrt(treatment_rate * (1 - treatment_rate) / len(treatment) + control_rate * (1 - control_rate) / len(control))
ci_lower = lift - 1.96 * se / control_rate
ci_upper = lift + 1.96 * se / control_rate
print(f"\n95%置信区间: [{ci_lower:.4f}, {ci_upper:.4f}]")

🚨 项目十：异常值检测
综合

综合运用统计方法和机器学习技术检测交易数据中的异常记录，包括欺诈交易、错误数据等。

异常检测 风控分析
涉及函数
zscore() IQR() IsolationForest() LocalOutlierFactor() 
import pandas as pd
import numpy as np
from scipy import stats
from sklearn.ensemble import IsolationForest
from sklearn.neighbors import LocalOutlierFactor

df = pd.read_csv('transactions.csv')

z_scores = np.abs(stats.zscore(df[['amount', 'quantity']]))
df['zscore_anomaly'] = (z_scores > 3).any(axis=1)
print(f"Z-Score 检测到 {df['zscore_anomaly'].sum()} 个异常")

Q1 = df[['amount', 'quantity']].quantile(0.25)
Q3 = df[['amount', 'quantity']].quantile(0.75)
IQR = Q3 - Q1
df['iqr_anomaly'] = ((df[['amount', 'quantity']] < (Q1 - 1.5 * IQR)) | 
                     (df[['amount', 'quantity']] > (Q3 + 1.5 * IQR))).any(axis=1)
print(f"IQR 检测到 {df['iqr_anomaly'].sum()} 个异常")

features = df[['amount', 'quantity', 'hour', 'day_of_week']]
iso_forest = IsolationForest(contamination=0.01, random_state=42)
df['iso_forest_anomaly'] = iso_forest.fit_predict(features) == -1

lof = LocalOutlierFactor(n_neighbors=20, contamination=0.01)
df['lof_anomaly'] = lof.fit_predict(features) == -1

df['anomaly_score'] = df['zscore_anomaly'].astype(int) + df['iqr_anomaly'].astype(int) + df['iso_forest_anomaly'].astype(int) + df['lof_anomaly'].astype(int)

df['final_anomaly'] = df['anomaly_score'] >= 2
print(f"\n综合检测到 {df['final_anomaly'].sum()} 个异常 ({df['final_anomaly'].mean()*100:.2f}%)")

anomalies = df[df['final_anomaly']]
anomalies.to_csv('detected_anomalies.csv', index=False)
print(f"\n异常记录已保存到 detected_anomalies.csv")
